private String lname;
private Car myFirstCar;
private List<Car> myOtherCars;

this.lname = lname;
}
public void setMyFirstCar(Car aCar) {
myFirstCar = aCar;
}
public void setOtherCars(List<Car> otherCars) {

}
@Override
public void aboutMe() {
System.out.println(

My First Car is : " + myFirstCar);
if(myOtherCars != null) {

}
}
}
